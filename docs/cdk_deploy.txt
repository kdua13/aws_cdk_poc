# reference taken from aws documentation - https://cdkworkshop.com/30-python.html

# steps to deploy cdk stack

# Create an empty directory on your system:
mkdir aws_cdk_poc && cd aws_cdk_poc

# command to create a new Python CDK project:
cdk init sample-app --language python

# activate the Virtualenv
source .venv/bin/activate

# install the required python modules
pip install -r requirements.txt

## define resources in the aws_cdk_poc_stack.py file

# command to synthesize the CloudFormation template for this code
cdk synth

# command to install the bootstrap stack 'CDKToolkit' into the environment
cdk bootstrap

# command to check diff
cdk list

# command to check diff
cdk diff

# command to deploy a CDK app
cdk deploy

# command to hotswap deploy during development only
cdk deploy --hotswap

# command to delete the above created stack
cdk destroy
